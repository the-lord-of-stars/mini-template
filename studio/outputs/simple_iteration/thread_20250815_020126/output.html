Question: What are the most common visualization techniques mentioned in the 'Abstract' and 'AuthorKeywords' columns, and how have they evolved over the years in relation to sensemaking research?

import pandas as pd
from collections import Counter
import matplotlib.pyplot as plt
import re

# Load the dataset
path = 'outputs/simple_iteration/thread_20250815_020126/dataset_selected.csv'
df = pd.read_csv(path)

# Function to extract visualization techniques from text
visualization_keywords = ['visualization', 'visual', 'tree', 'graph', 'plot', 'chart', 'diagram', 'map']

def extract_visualization_techniques(text):
    if pd.isnull(text):
        return []
    words = re.findall(r'\b\w+\b', text.lower())
    return [word for word in words if word in visualization_keywords]

# Extract techniques from Abstract and AuthorKeywords
abstract_techniques = df['Abstract'].apply(extract_visualization_techniques)
keywords_techniques = df['AuthorKeywords'].apply(extract_visualization_techniques)

# Combine techniques from both columns
all_techniques = abstract_techniques + keywords_techniques

# Flatten the list of lists and count occurrences
all_techniques_flat = [technique for sublist in all_techniques for technique in sublist]
technique_counts = Counter(all_techniques_flat)

# Print the most common visualization techniques
print("### Begin of facts")
print("Most common visualization techniques:")
for technique, count in technique_counts.most_common(5):
    print(f"{technique}: {count}")
print("### End of facts")

# Analyze evolution over the years
techniques_over_years = {}
for year, group in df.groupby('Year'):
    year_techniques = group['Abstract'].apply(extract_visualization_techniques) + group['AuthorKeywords'].apply(extract_visualization_techniques)
    year_techniques_flat = [technique for sublist in year_techniques for technique in sublist]
    techniques_over_years[year] = Counter(year_techniques_flat)

# Plot the evolution of the top techniques
plt.figure(figsize=(10, 6))
for technique in technique_counts.most_common(3):
    technique_name = technique[0]
    years = sorted(techniques_over_years.keys())
    counts = [techniques_over_years[year][technique_name] for year in years]
    plt.plot(years, counts, label=technique_name)

plt.title('Evolution of Visualization Techniques in Sensemaking Research')
plt.xlabel('Year')
plt.ylabel('Mentions')
plt.legend()
plt.grid(True)
plt.show()

Iteration 2

Question: How has the integration of collaborative frameworks, such as 'computer-supported cooperative work', with visualization techniques in sensemaking research evolved over the years, as reflected in the 'Abstract' and 'AuthorKeywords' columns?
ðŸ“Š Analysis Code (click to expand)
import pandas as pd
import matplotlib.pyplot as plt
from collections import Counter

# Load the dataset
df = pd.read_csv('outputs/simple_iteration/thread_20250815_020126/dataset_selected.csv')

# Filter relevant keywords and terms
keywords_of_interest = ['collaboration', 'computer-supported cooperative work', 'visualization', 'sensemaking']

# Extract years and keywords
years = df['Year']
keywords = df['AuthorKeywords'].fillna('')

# Count occurrences of keywords over the years
keyword_trends = {key: [] for key in keywords_of_interest}
for year in sorted(years.unique()):
    year_keywords = ' '.join(keywords[years == year]).lower()
    keyword_counts = Counter(year_keywords.split(','))
    for key in keywords_of_interest:
        keyword_trends[key].append(keyword_counts.get(key, 0))

# Plot the trends
plt.figure(figsize=(10, 6))
for key, trend in keyword_trends.items():
    plt.plot(sorted(years.unique()), trend, label=key)

plt.title('Trends in Collaborative Frameworks and Visualization in Sensemaking Research')
plt.xlabel('Year')
plt.ylabel('Number of Papers')
plt.legend()
plt.grid(True)
plt.show()

# Print facts
print("### Begin of facts")
for key, trend in keyword_trends.items():
    print(f"Keyword '{key}' appeared in {sum(trend)} papers over the years.")
print("### End of facts")
