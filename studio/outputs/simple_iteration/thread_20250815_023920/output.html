years = df['Year']

# Get the most frequently mentioned authors
most_frequent_authors = author_counts.most_common(5)

# Analyze the evolution of sensemaking research from 2010 to 2012
sensemaking_papers = df[df['Title'].str.contains('sensemaking', case=False) | df['Abstract'].str.contains('sensemaking', case=False)]
sensemaking_authors = sensemaking_papers['AuthorNames'].str.split(';').explode()
sensemaking_years = sensemaking_papers['Year']

# Count the frequency of each author in sensemaking papers
sensemaking_author_counts = Counter(sensemaking_authors)

print("### Begin of facts")
print("Most frequently mentioned authors:")
for author, count in most_frequent_authors:
    print(f"{author}: {count} times")

    authors_in_year = sensemaking_authors[sensemaking_years == year]
