Question: How do the citation counts of sensemaking papers correlate with the number of downloads from Xplore, and what does this indicate about the impact and reach of these papers?

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Load the dataset
df = pd.read_csv('outputs/simple_iteration/thread_20250815_033304/dataset_selected.csv')

# Filter papers related to sensemaking
df_sensemaking = df[df['AuthorKeywords'].str.contains('sensemaking', case=False, na=False)]

# Calculate correlation between citation counts and downloads
correlation = df_sensemaking['CitationCount_CrossRef'].corr(df_sensemaking['Downloads_Xplore'])

# Plot the relationship
plt.figure(figsize=(10, 6))
sns.scatterplot(x='CitationCount_CrossRef', y='Downloads_Xplore', data=df_sensemaking)
plt.title('Correlation between Citation Counts and Downloads for Sensemaking Papers')
plt.xlabel('Citation Count (CrossRef)')
plt.ylabel('Downloads (Xplore)')
plt.grid(True)
plt.show()

# Print the facts
print("### Begin of facts")
print(f"Number of sensemaking papers: {len(df_sensemaking)}")
print(f"Correlation between citation counts and downloads: {correlation:.2f}")
print("### End of facts")
