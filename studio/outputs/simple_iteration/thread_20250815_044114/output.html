Iteration 1

Question: How have the authorship patterns in sensemaking research evolved over the years, and are there any notable collaborations or recurring authors in this field?

import pandas as pd
from collections import Counter

# Load the dataset
df = pd.read_csv('outputs/simple_iteration/thread_20250815_044114/dataset_selected.csv')

# Filter papers related to sensemaking
df_sensemaking = df[df['AuthorKeywords'].str.contains('sensemaking', case=False, na=False)]

# Extract authors and years
authors_years = df_sensemaking[['Year', 'AuthorNames']]

# Split authors and create a list of (year, author) tuples
author_year_list = []
for _, row in authors_years.iterrows():
    year = row['Year']
    authors = row['AuthorNames'].split(';')
    for author in authors:
        author_year_list.append((year, author.strip()))

# Count occurrences of each author
author_counts = Counter([author for _, author in author_year_list])

# Find the most common authors
top_authors = author_counts.most_common(5)

# Find collaborations by creating a set of author pairs for each paper
author_pairs = []
for _, row in authors_years.iterrows():
    authors = row['AuthorNames'].split(';')
    authors = [author.strip() for author in authors]
    for i in range(len(authors)):
        for j in range(i + 1, len(authors)):
            author_pairs.append(tuple(sorted((authors[i], authors[j]))))

# Count collaborations
author_collaborations = Counter(author_pairs)

# Find the most common collaborations
top_collaborations = author_collaborations.most_common(5)

# Print the facts
print("### Begin of facts")
print(f"Total sensemaking papers: {len(df_sensemaking)}")
print("Top 5 authors in sensemaking research:")
for author, count in top_authors:
    print(f"{author}: {count} papers")
print("Top 5 collaborations in sensemaking research:")
for (author1, author2), count in top_collaborations:
    print(f"{author1} & {author2}: {count} collaborations")
print("### End of facts")
