Iteration 1

Question: What are the common themes or topics in the abstracts from 2010 to 2012 that indicate a shift in the research focus on sensemaking?
ðŸ“Š Analysis Code (click to expand)
import pandas as pd
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.decomposition import LatentDirichletAllocation

# Load the dataset
df = pd.read_csv('outputs/simple_iteration/thread_20250815_044431/dataset_selected.csv')

# Filter the dataset for the years 2010 to 2012
df_filtered = df[(df['Year'] >= 2010) & (df['Year'] <= 2012)]

# Extract abstracts
documents = df_filtered['Abstract'].values

# Vectorize the text data
vectorizer = CountVectorizer(stop_words='english')
X = vectorizer.fit_transform(documents)

# Apply LDA to find common topics
lda = LatentDirichletAllocation(n_components=3, random_state=42)
lda.fit(X)

# Get the top words for each topic
words = vectorizer.get_feature_names_out()
topics = []
for topic_idx, topic in enumerate(lda.components_):
    top_words = [words[i] for i in topic.argsort()[:-6:-1]]
    topics.append("Topic {}: ".format(topic_idx + 1) + ", ".join(top_words))

# Print the topics
print("### Begin of facts")
print("Common themes or topics in abstracts from 2010 to 2012:")
for topic in topics:
    print(topic)
print("### End of facts")

Question: What are the key differences in the use of 'visual analytics' in sensemaking research between 2010 and 2012, as reflected in the abstracts and 'AuthorKeywords'?

import pandas as pd
from collections import Counter

# Load the dataset
path = 'outputs/simple_iteration/thread_20250815_044431/dataset_selected.csv'
df = pd.read_csv(path)

# Filter the dataset for the years 2010 and 2012
filtered_df = df[df['Year'].isin([2010, 2012])]

# Extract keywords and abstracts for each year
keywords_2010 = ' '.join(filtered_df[filtered_df['Year'] == 2010]['AuthorKeywords']).lower().split(', ')
keywords_2012 = ' '.join(filtered_df[filtered_df['Year'] == 2012]['AuthorKeywords']).lower().split(', ')

abstracts_2010 = ' '.join(filtered_df[filtered_df['Year'] == 2010]['Abstract']).lower().split()
abstracts_2012 = ' '.join(filtered_df[filtered_df['Year'] == 2012]['Abstract']).lower().split()

# Count the frequency of keywords and words in abstracts
keywords_count_2010 = Counter(keywords_2010)
keywords_count_2012 = Counter(keywords_2012)

abstracts_count_2010 = Counter(abstracts_2010)
abstracts_count_2012 = Counter(abstracts_2012)

# Get the most common keywords and words in abstracts
common_keywords_2010 = keywords_count_2010.most_common(5)
common_keywords_2012 = keywords_count_2012.most_common(5)

common_abstracts_2010 = abstracts_count_2010.most_common(5)
common_abstracts_2012 = abstracts_count_2012.most_common(5)

# Print the results
print("### Begin of facts")
print("Most common keywords in 2010:", common_keywords_2010)
print("Most common keywords in 2012:", common_keywords_2012)
print("Most common words in abstracts in 2010:", common_abstracts_2010)
print("Most common words in abstracts in 2012:", common_abstracts_2012)
print("### End of facts")
